local MainTab = Window:CreateTab("FarmLevel", 4483362458)

local function getNPC()
   for _, child in ipairs(workspace.NPCs:GetChildren()) do
       if child:FindFirstChild("Quests") then
           return child
       end
   end
   return nil
end

_G.FarmLevel = false

MainTab:CreateToggle({
   Name = "FarmAuto",
   CurrentValue = false,
   Flag = "FarmAutoToggle",
   Callback = function(Value)
      _G.FarmLevel = Value
      if _G.FarmLevel then
         spawn(function()
            while _G.FarmLevel do
               task.wait(0.3)
               pcall(function()
                   local localPlayer = game:GetService("Players").LocalPlayer
                   local localChar = localPlayer.Character
                   local localRoot = localChar and localChar:FindFirstChild("HumanoidRootPart")
                   if localRoot then
                       local args1 = {
                           [1] = "Iron Sword",
                           [2] = "OnActivated",
                           [3] = {
                               [1] = localChar:FindFirstChild("Iron Sword"),
                               [2] = 1
                           }
                       }
                       game:GetService("ReplicatedStorage").Modules.Shared.Event.Remotes.ClientToServerCallback:FireServer(unpack(args1))
                       
                       local mobsFolder = workspace:FindFirstChild("Mobs")
                       if mobsFolder then
                           for _, mob in ipairs(mobsFolder:GetChildren()) do
                               local humanoid = mob:FindFirstChild("Humanoid")
                               if humanoid and humanoid.Health > 0 then
                                   local mobRoot = mob:FindFirstChild("HumanoidRootPart") or mob:FindFirstChildWhichIsA("Part")
                                   if mobRoot then
                                       local distance = (mobRoot.Position - localRoot.Position).Magnitude
                                       if distance <= 45 then
                                           local args3 = {
                                               [1] = mob,
                                               [2] = "Melee"
                                           }
                                           for i = 1, 15 do
                                               game:GetService("ReplicatedStorage").Modules.Shared.Event.Remotes.RequestMeleeHitbox:FireServer(unpack(args3))
                                           end
                                       end
                                   end
                               end
                           end
                       end
                   end
               end)
            end
         end)
      end
   end,
})

_G.MineAuto = false

MainTab:CreateToggle({
   Name = "MineAuto",
   CurrentValue = false,
   Flag = "MineAutoToggle",
   Callback = function(Value)
      _G.MineAuto = Value
      if _G.MineAuto then
         spawn(function()
            while _G.MineAuto do
               task.wait(0.5)
               pcall(function()
                   local localPlayer = game.Players.LocalPlayer
                   local character = localPlayer.Character
                   local localRoot = character and character:FindFirstChild("HumanoidRootPart")
                   if localRoot then
                       local propsFolder = workspace:FindFirstChild("Props")
                       if propsFolder then
                           local closestStone = nil
                           for _, prop in ipairs(propsFolder:GetChildren()) do
                               if string.find(prop.Name, "Stone") or prop.Name == "Stone Chunk2" then
                                   local position = nil
                                   if prop:IsA("Model") and prop.PrimaryPart then
                                       position = prop.PrimaryPart.Position
                                   elseif prop:IsA("Part") then
                                       position = prop.Position
                                   end
                                   if position then
                                       local distance = (position - localRoot.Position).Magnitude
                                       if distance <= 25 then
                                           closestStone = prop
                                           break
                                       end
                                   end
                               end
                           end
                           if closestStone then
                               local args = {
                                   [1] = closestStone,
                                   [2] = "Pickaxe"
                               }
                               game:GetService("ReplicatedStorage").Modules.Shared.Event.Remotes.RequestMeleeHitbox:FireServer(unpack(args))
                           end
                       end
                   end
               end)
            end
         end)
      end
   end,
})

local QuestSection = MainTab:CreateSection("Quest Auto")

_G.QuestLv1_10 = false
_G.QuestLv20_30 = false

local NpcPosition = CFrame.new(1028.89368, 246.434509, 2067.24268)
local FarmPosition = CFrame.new(1076.27795, 246.434509, 2151.44067)

MainTab:CreateToggle({
   Name = "QuestLv1-10",
   CurrentValue = false,
   Flag = "Quest1_10",
   Callback = function(Value)
      _G.QuestLv1_10 = Value
      if Value then
         spawn(function()
            while _G.QuestLv1_10 do
               pcall(function()
                   local localPlayer = game.Players.LocalPlayer
                   local character = localPlayer.Character
                   local localRoot = character and character:FindFirstChild("HumanoidRootPart")
                   if localRoot then
                       localRoot.CFrame = NpcPosition
                       task.wait(0.5)
                       local npc = getNPC()
                       local questObj = npc and npc:FindFirstChild("Quests") and npc.Quests:FindFirstChild("Trainer_Marcus")
                       if questObj then
                           local args1 = {
                               [1] = questObj
                           }
                           game:GetService("ReplicatedStorage").Modules.Shared.Event.Remotes.TalkToNPC:FireServer(unpack(args1))
                           task.wait(0.2)
                           local args2 = {
                               [1] = "rookie_training_2"
                           }
                           game:GetService("ReplicatedStorage").Modules.Shared.Event.Remotes.ExitQuest:FireServer(unpack(args2))
                           task.wait(0.2)
                           local args3 = {
                               [1] = "rookie_training_2"
                           }
                           local questResult = game:GetService("ReplicatedStorage").Modules.Shared.Event.Remotes.QuestStart:InvokeServer(unpack(args3))
                           if questResult ~= false then
                               task.wait(0.5)
                               localRoot.CFrame = FarmPosition
                               task.wait(8.5)
                           end
                       end
                   end
               end)
               task.wait(1)
            end
         end)
      end
   end,
})
