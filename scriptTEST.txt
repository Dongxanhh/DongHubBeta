   LoadingTitle = "Loading DongxanhHub...",
   LoadingSubtitle = "by Dong",
   ConfigurationSaving = {
      Enabled = true,
      FolderName = "DongHubConfig",
      FileName = "MainHub"
   },
   KeySystem = false
})

local MainTab = Window:CreateTab("FarmLevel", 4483362458)

_G.FarmLevel = false

MainTab:CreateToggle({
   Name = "Farm Level",
   CurrentValue = false,
   Flag = "FarmLevelToggle",
   Callback = function(Value)
      _G.FarmLevel = Value
      
      if _G.FarmLevel then
         spawn(function()
            while _G.FarmLevel do
               task.wait(0.05) 
               
               pcall(function()
                   local localChar = game:GetService("Players").LocalPlayer.Character
                   
                   local args1 = {
                       [1] = localChar,
                       [2] = "Melee"
                   }
                   game:GetService("ReplicatedStorage").Modules.Shared.Event.Remotes.RequestMeleeHitbox:FireServer(unpack(args1))
                   
                   local mobsFolder = workspace:FindFirstChild("Mobs")
                   if mobsFolder then
                       for _, mob in ipairs(mobsFolder:GetChildren()) do
                           local args3 = {
                               [1] = mob,
                               [2] = "Melee"
                           }
                           game:GetService("ReplicatedStorage").Modules.Shared.Event.Remotes.RequestMeleeHitbox:FireServer(unpack(args3))
                       end
                   end
               end)
               
               print("Farm...")
            end
         end)
      end
   end,
})

_G.QuestAuto = false

-- Tọa độ đã được rút gọn chỉ giữ lại X, Y, Z để tránh lỗi khi copy qua GitHub
local NpcPosition = CFrame.new(1028.89368, 246.434509, 2067.24268)
local FarmPosition = CFrame.new(1076.27795, 246.434509, 2151.44067)

MainTab:CreateToggle({
   Name = "Quest Auto",
   CurrentValue = false,
   Flag = "QuestAutoToggle",
   Callback = function(Value)
      _G.QuestAuto = Value
      
      if _G.QuestAuto then
         spawn(function()
            while _G.QuestAuto do
               pcall(function()
                   local localPlayer = game.Players.LocalPlayer
                   local character = localPlayer.Character
                   local humanoidRootPart = character and character:FindFirstChild("HumanoidRootPart")
                   
                   if humanoidRootPart then
                       local originalCFrame = humanoidRootPart.CFrame
                       
                       humanoidRootPart.CFrame = NpcPosition
                       task.wait(0.5)
                       
                       local targetNpc = nil
                       local minDist = 5
                       for _, v in pairs(workspace.NPCs:GetChildren()) do
                           local part = v:FindFirstChild("HumanoidRootPart") or v:FindFirstChildWhichIsA("Part")
                           if part then
                               local dist = (part.Position - humanoidRootPart.Position).Magnitude
                               if dist < minDist then
                                   minDist = dist
                                   targetNpc = v
                                   break
                               end
                           end
                       end
                       
                       if targetNpc then
                           local args1 = {
                               [1] = targetNpc
                           }
                           game:GetService("ReplicatedStorage").Modules.Shared.Event.Remotes.TalkToNPCStart:FireServer(unpack(args1))
                           
                           task.wait(0.5)
                           
                           local args2 = {
                               [1] = "rookie_training_2"
                           }
                           game:GetService("ReplicatedStorage").Modules.Shared.Event.Remotes.QuestStart:InvokeServer(unpack(args2))
                           
                           task.wait(0.5)
                           
                           local args3 = {
                               [1] = targetNpc
                           }
                           game:GetService("ReplicatedStorage").Modules.Shared.Event.Remotes.TalkToNPC:FireServer(unpack(args3))
                       end
                       
                       task.wait(1.0)
                       
                       humanoidRootPart.CFrame = FarmPosition
                   end
               end)
               
               task.wait(5)
            end
         end)
      end
   end,
})
