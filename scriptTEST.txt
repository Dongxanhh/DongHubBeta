local LocalizationService = game:GetService("LocalizationService")
local Rayfield = loadstring(game:HttpGet('https://raw.githubusercontent.com/SiriusSoftwareLtd/Rayfield/main/source'))()

local Window = Rayfield:CreateWindow({
   Name = "DongxanhHub | Beta0.1 Menu",
   LoadingTitle = "Loading DongxanhHub...",
   LoadingSubtitle = "by Dong",
   ConfigurationSaving = {
      Enabled = true,
      FolderName = "DongHubConfig",
      FileName = "MainHub"
   },
   KeySystem = false
})

local MainTab = Window:CreateTab("FarmLevel", 4483362458)

_G.FarmLevel = false

MainTab:CreateToggle({
   Name = "FarmAuto",
   CurrentValue = false,
   Flag = "FarmAutoToggle",
   Callback = function(Value)
      _G.FarmLevel = Value
      
      if _G.FarmLevel then
         spawn(function()
            while _G.FarmLevel do
               task.wait(0.3)
               
               pcall(function()
                   local localPlayer = game:GetService("Players").LocalPlayer
                   local localChar = localPlayer.Character
                   local localRoot = localChar and localChar:FindFirstChild("HumanoidRootPart")
                   
                   if localRoot then
                       local args1 = {
                           [1] = localChar,
                           [2] = "Melee"
                       }
                       game:GetService("ReplicatedStorage").Modules.Shared.Event.Remotes.RequestMeleeHitbox:FireServer(unpack(args1))
                       
                       local mobsFolder = workspace:FindFirstChild("Mobs")
                       if mobsFolder then
                           for _, mob in ipairs(mobsFolder:GetChildren()) do
                               local humanoid = mob:FindFirstChild("Humanoid")
                               if humanoid and humanoid.Health > 0 then
                                   local mobRoot = mob:FindFirstChild("HumanoidRootPart") or mob:FindFirstChildWhichIsA("Part")
                                   if mobRoot then
                                       local distance = (mobRoot.Position - localRoot.Position).Magnitude
                                       if distance <= 45 then
                                           local args3 = {
                                               [1] = mob,
                                               [2] = "Melee"
                                           }
                                           game:GetService("ReplicatedStorage").Modules.Shared.Event.Remotes.RequestMeleeHitbox:FireServer(unpack(args3))
                                       end
                                   end
                               end
                           end
                       end
                   end
               end)
            end
         end)
      end
   end,
})

local MiscTab = Window:CreateTab("Misc", 4483345998)

MiscTab:CreateButton({
   Name = "Close Menu",
   Callback = function()
      Rayfield:Destroy()
   end,
})

Rayfield:Notify({
   Title = "DongxanhHub On!",
   Content = "",
   Duration = 5,
   Image = 4483345998,
})
